<?php 
$a = 5;
$b = '05';
var_dump($a == $b); // PHP интерпретировал текст в число
var_dump((int) '012345'); // при интерпритации из строки в число убрал 0
var_dump((float) 123.0 === (int) 123.0); // float работает с числами с плавующей точкой, а int с целыми числами
var_dump((int) 0 === (int) 'Hello? word!'); // строка будет интерпритирована в число, тк в строке чисел нет она будет ровна 0
?>
